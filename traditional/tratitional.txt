import cv2
import argparse
import os
import numpy as np

# 调用统一接口
from traditional.infer import TraditionalSegmentor
from deep_learning.infer import DeepLearningSegmentor
from utils.visualizer import show_resizable  # 交互式显示
from utils.labelme2mask import convert_labelme_to_mask

def build_segmentor(method: str):
    if method == "traditional":
        return TraditionalSegmentor()
    elif method == "deep_learning":
        weight_path = "deep_learning/unet.pth"
        if not os.path.exists(weight_path):
            raise FileNotFoundError(
                f"Deep learning model weights not found: {weight_path}. "
                "请先运行 deep_learning/train.py 训练模型"
            )
        return DeepLearningSegmentor(weight_path)
    else:
        raise ValueError(f"Unknown method: {method}")


def main():
    parser = argparse.ArgumentParser(
        description="Passable Area Segmentation System"
    )
    parser.add_argument(
        "--method",
        type=str,
        default="traditional",
        choices=["traditional", "deep_learning"],
        help="选择分割方法"
    )
    parser.add_argument(
        "--image",
        type=str,
        default=None,
        help="单张图片路径（可选）"
    )
    parser.add_argument(
        "--show_overlay",
        action="store_true",
        help="显示结果叠加图"
    )
    parser.add_argument(
        "--process_all",
        action="store_true",
        help="批量处理 data/raw 目录下所有图片"
    )
    args = parser.parse_args()

    # 默认测试
    if not args.image and not args.process_all:
        default_image = "data/raw/test1.jpg"
        if not os.path.exists(default_image):
            print(f"⚠️ 默认测试图片不存在：{default_image}")
            print("请使用 --image 指定图片，或放一张 test1.jpg 到 data/raw/")
            return
        print(f"⚠️ 未指定参数，默认测试 {default_image}")
        args.image = default_image

    # 构建分割器
    segmentor = build_segmentor(args.method)

    # -------------------- 单张处理 --------------------
    if args.image:
        image = cv2.imread(args.image)
        if image is None:
            raise FileNotFoundError(f"Image not found: {args.image}")

        mask = segmentor.segment(image)

        # 显示原图
        show_resizable("Input Image", image)

        # 显示 mask
        mask_color = cv2.cvtColor(mask, cv2.COLOR_GRAY2BGR)
        show_resizable("Mask", mask_color)

        # 显示 overlay
        if args.show_overlay:
            overlay = image.copy()
            overlay[mask > 0] = (0, 255, 0)
            blended = cv2.addWeighted(image, 0.7, overlay, 0.3, 0)
            show_resizable("Overlay", blended)

    # -------------------- 批量处理 --------------------
    elif args.process_all:
        raw_dir = "data/raw"
        output_dir = "output_masks"
        os.makedirs(output_dir, exist_ok=True)

        for fname in sorted(os.listdir(raw_dir)):
            img_path = os.path.join(raw_dir, fname)
            image = cv2.imread(img_path)
            if image is None:
                continue
            mask = segmentor.segment(image)
            save_path = os.path.join(output_dir, fname)
            cv2.imwrite(save_path, mask)
            print(f"Saved mask: {save_path}")


if __name__ == "__main__":
    main()
